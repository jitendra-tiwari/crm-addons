//count using mesure
LeadData = CALCULATE (
    COUNT ( 'inhousecrm lead'[lead_date] ),
    FILTER (
        ALL ('inhousecrm lead'),
       YEAR( 'inhousecrm lead'[lead_date])=YEAR(NOW()) && MONTH('inhousecrm lead'[lead_date])=MONTH(NOW())
    )
)


//rowcount by mesure
